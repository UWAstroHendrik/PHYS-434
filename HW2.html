4) Repeat 2, and graph the average of 10 dice. Is this is a Gaussian distribution? Explain in depth.¶

In [76]:
def average_probability_of_10_dice(throws, width=2):
    average_probability = np.copy(probability_sum_of_two)
    for i in range(throws):
        average_probability = np.convolve(probability_sum_of_two, average_probability)        
    possible_average = np.linspace(1, 6, len(average_probability))
    plt.vlines(possible_average, 0, average_probability, linestyles ="dotted", colors = "k", linewidth=width);
    plt.xlabel("Possible Average")
    plt.ylabel("Probability")
    plt.xlim(1, 6)
    plt.ylim(0, 0.08)
    return possible_average, average_probability
